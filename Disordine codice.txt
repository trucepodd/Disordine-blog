    // --- FUNZIONE PRINCIPALE DI NAVIGAZIONE ---
    function showPage(pageName) {
        // Nascondi tutto
        personalPage.classList.add('hidden');
        blogPage.classList.add('hidden');
        singleArticlePage.classList.add('hidden');

        // Scroll automatico in cima

        }

    // --- RENDERIZZA GLI ARTICOLI (LISTA) ---
    function renderArticles() {
        articlesContainer.innerHTML = ''; 

        if (articoli.length === 0) {
            articlesContainer.innerHTML = '<p style="text-align:center;">No posts yet.</p>';
            return;
        }

        articoli.forEach(art => {
            const articleDiv = document.createElement('div');
            articleDiv.className = 'article-item';
            
            // Nella lista usiamo il sottotitolo come anteprima

                <div class="article-title" onclick="openArticle(${art.id})">${art.titolo}</div>

            `;
            articlesContainer.appendChild(articleDiv);
        });
    }

    // --- APRE UN ARTICOLO SINGOLO ---
    function openArticle(id) {
        const art = articoli.find(a => a.id === id);
        if (!art) return;

        blogPage.classList.add('hidden');
        singleArticlePage.classList.remove('hidden');
        window.scrollTo(0, 0);

        document.getElementById('post-title').innerText = art.titolo;
        document.getElementById('post-subtitle').innerText = art.sottotitolo; // Ora mostra il sottotitolo
        document.getElementById('post-date').innerText = art.data;
        document.getElementById('post-content').innerText = art.contenuto;
    }
